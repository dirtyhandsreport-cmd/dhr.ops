                </div>
              </>
            )}
          </div>
        )}
        {listings.map(l=>(
          <div key={l.id} className="scard">
            <div style={{display:"flex",justifyContent:"space-between",alignItems:"flex-start",marginBottom:8}}>
              <div>
                <div style={{fontSize:15,fontWeight:800,color:"#fff",fontFamily:"'Barlow Condensed',sans-serif"}}>{l.title}</div>
                <div style={{fontSize:11,color:"rgba(255,255,255,.4)",marginTop:2,fontFamily:"'Barlow',sans-serif"}}>{l.company} · {l.location} · {l.pay}</div>
              </div>
              <div style={{display:"flex",flexDirection:"column",alignItems:"flex-end",gap:4}}>
                <span style={{fontSize:11,color:"#22c55e",fontWeight:800,fontFamily:"'Barlow Condensed',sans-serif"}}>{l.clicks} clicks</span>
                <button style={{background:"rgba(255,255,255,.06)",border:"none",color:"rgba(255,255,255,.3)",cursor:"pointer",fontSize:11,width:24,height:24,borderRadius:12,display:"flex",alignItems:"center",justifyContent:"center"}} onClick={()=>setListings(p=>p.filter(x=>x.id!==l.id))}>✕</button>

              <button className="gbtn" style={{fontSize:10,padding:"8px 12px"}} onClick={()=>generateListing(l)}>✍ GENERATE POST</button>
              <button className="gbtn" style={{fontSize:10,padding:"8px 12px"}} onClick={()=>setListings(p=>p.map(x=>x.id===l.id?{...x,clicks:x.clicks+1}:x))}>+ CLICK</button>

          <div className="scard" style={{border:"1px solid rgba(240,90,26,.3)"}}>
            <div style={{display:"flex",flexDirection:"column",gap:7}}>
              <input className="tinput" placeholder="Job title" value={newL.title} onChange={e=>setNewL(p=>({...p,title:e.target.value}))}/>
              <input className="tinput" placeholder="Company" value={newL.company} onChange={e=>setNewL(p=>({...p,company:e.target.value}))}/>
              <input className="tinput" placeholder="Location" value={newL.location} onChange={e=>setNewL(p=>({...p,location:e.target.value}))}/>
              <input className="tinput" placeholder="Pay / rate" value={newL.pay} onChange={e=>setNewL(p=>({...p,pay:e.target.value}))}/>
              <input className="tinput" placeholder="Notes" value={newL.notes} onChange={e=>setNewL(p=>({...p,notes:e.target.value}))}/>
              <div style={{display:"flex",gap:6}}>
                <button className="sbtn" style={{flex:1}} onClick={()=>{if(!newL.title)return;setListings(p=>[...p,{...newL,id:Date.now(),clicks:0,status:"ACTIVE",posted:new Date().toLocaleDateString("en-US",{month:"short",day:"numeric",year:"numeric"})}]);setNewL({title:"",company:"",location:"",pay:"",notes:""});setAdding(false);}}>ADD LISTING</button>
                <button className="gbtn" onClick={()=>setAdding(false)}>CANCEL</button>
              </div>
            </div>
          </div>
        ):<button className="gbtn" style={{width:"100%",marginTop:4,fontSize:11}} onClick={()=>setAdding(true)}>+ NEW LISTING</button>}
      </div>
    </div>
  );
}

export default function DHRFactory(){
  const [screen,setScreen]=useState("news");
  const [postHistory,setPostHistory]=useState([]);
  const [hookLibrary,setHookLibrary]=useState(HOOK_LIB_DEFAULT);
  const [genFor,setGenFor]=useState(null);
  const [schedFor,setSchedFor]=useState(null);
  const [toast,setToast]=useState("");
  const showToast=msg=>{setToast(msg);setTimeout(()=>setToast(""),2200);};
  const saveHook=(text,removeIdx=null)=>{
    if(removeIdx!==null){setHookLibrary(p=>p.filter((_,i)=>i!==removeIdx));return;}
    if(text&&!hookLibrary.includes(text))setHookLibrary(p=>[text,...p]);
    showToast("✓ HOOK SAVED");
  };

  const NAV=[
    {k:"news",     icon:"⛽", label:"NEWS"},
    {k:"write",    icon:"✍",  label:"WRITE"},
    {k:"sponsors", icon:"💰", label:"SPONSORS"},
    {k:"jobs",     icon:"🔧", label:"JOBS"},
    {k:"brand",    icon:"🎨", label:"BRAND"},
  ];

  return(
    <div style={{minHeight:"100vh",background:"#0a0a0f",color:"#f0ece4",fontFamily:"'Barlow Condensed','Arial Narrow',Arial,sans-serif",paddingBottom:72}}>
      <style>{CSS}</style>
      {toast&&<div className="toast">{toast}</div>}
      {genFor&&<PostModal headline={genFor} onClose={()=>setGenFor(null)} hookLibrary={hookLibrary} onSaveHook={saveHook} onSave={item=>{setPostHistory(p=>[item,...p]);showToast("✓ SAVED TO HISTORY");}}/>}
      {schedFor&&<SchedModal headline={schedFor} onClose={()=>setSchedFor(null)} onSave={()=>showToast("✓ ADDED TO CALENDAR")}/>}
      <PriceTicker/>
      <div className="screen" key={screen}>
        {screen==="news"     && <NewsScreen    onGen={setGenFor} onSchedule={setSchedFor} onToast={showToast}/>}
        {screen==="write"    && <WriteScreen   postHistory={postHistory} onToast={showToast} hookLibrary={hookLibrary} onSaveHook={saveHook}/>}
        {screen==="sponsors" && <SponsorScreen onToast={showToast}/>}
        {screen==="jobs"     && <JobsScreen    onToast={showToast}/>}
        {screen==="brand"    && <BrandScreen/>}
      </div>
      <nav className="nav-bar">
        {NAV.map(n=>(
          <button key={n.k} className={`nav-btn ${screen===n.k?"on":""}`} onClick={()=>setScreen(n.k)}>
            <div className="nav-icon-bg"><span className="nav-icon">{n.icon}</span></div>
            {n.label}
          </button>
        ))}
      </nav>
    </div>
  );
}
